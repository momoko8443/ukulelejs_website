<select uku-selected="ex10Ctrl.selectedOption|value" uku-onchange="ex10Ctrl.selectedOptionChanged()">
  <option uku-repeat="option in ex10Ctrl.options"
  			uku-value="option.value" uku-data-item="option">{{option.name}}</option>
</select>	
<select uku-selected="ex10Ctrl.selectedChildOption|name">
  <option uku-repeat="child in ex10Ctrl.selectedOption.children"
  			uku-value="child.name" uku-data-item="child">{{child.name}}</option>
</select>
<label>{{ex10Ctrl.selectedOption.name}}</label>
<label>{{ex10Ctrl.selectedChildOption.name}}</label>
					    		
		
			
function Example10Ctrl() {		 
	this.options = [
		{"name":"Kamaka","value":"kamaka","children":[
			{"name":"HF1"},{"name":"HF2"},{"name":"HF3"}
		]},
		{"name":"Koaloha","value":"koaloha","children":[
			{"name":"KSM"},{"name":"KCM"},{"name":"KTM"}
		]},
		{"name":"Kanilea","value":"kanilea","children":[
			{"name":"K1S"},{"name":"K2C"},{"name":"K1T"}
		]},
		{"name":"Koolau ","value":"koolau","children":[
			{"name":"CS Tenor"},{"name":"Model 100"},{"name":"Tenor Deluxe"}
		]}
	];
	this.selectedOption = this.options[2];
	this.selectedChildOption = this.selectedOption.children[0];
	
	this.selectedOptionChanged = function(){
		this.selectedChildOption = this.selectedOption.children[0];	
		uku.refresh();
	};
}
